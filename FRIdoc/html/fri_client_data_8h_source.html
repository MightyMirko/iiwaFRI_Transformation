161  //******************************************************************************
162  static const FriIOValue& getIOValue(const FRIMonitoringMessage* message, const char* name,
163  const FriIOType ioType)
164  {
165  if(message != NULL && message->has_monitorData == true)
166  {
167  const MessageMonitorData& monData = message->monitorData;
168  const bool analogValue = (ioType == FriIOType_ANALOG);
169  const bool digitalValue = (ioType == FriIOType_DIGITAL | ioType == FriIOType_BOOLEAN);
170  for(size_t i = 0; i < monData.readIORequest_count; i++)
171  {
172  const FriIOValue& ioValue = monData.readIORequest[i];
173  if(strcmp(name, ioValue.name) == 0)
174  {
175  if(ioValue.type == ioType &&
176  ioValue.has_digitalValue == digitalValue &&
177  ioValue.has_analogValue == analogValue)
178  {
179  return ioValue;
180  }
181 
182  const char* ioTypeName;
183  switch(ioType)
184  {
185  case FriIOType_ANALOG: ioTypeName = "analog value"; break;
186  case FriIOType_DIGITAL: ioTypeName = "digital value"; break;
187  case FriIOType_BOOLEAN: ioTypeName = "boolean"; break;
188  default: ioTypeName = "?"; break;
189  }
190 
191  throw FRIException("IO %s is not of type %s.", name, ioTypeName);
192  }
193  }
194  }
195 
196  throw FRIException("Could not locate IO %s in monitor message.", name);
197  }

199  //******************************************************************************
200  static FriIOValue& setIOValue(FRICommandMessage* message, const char* name,
201  const FRIMonitoringMessage* monMessage, const FriIOType ioType)
202  {
203  MessageCommandData& cmdData = message->commandData;
204  const size_t maxIOs = sizeof(cmdData.writeIORequest) / sizeof(cmdData.writeIORequest[0]);
205  if(cmdData.writeIORequest_count < maxIOs)
206  {
207  // call getter which will raise an exception if the output doesn't exist
208  // or is of wrong type.
209  if(getIOValue(monMessage, name, ioType).direction != FriIODirection_OUTPUT)
210  {
211  throw FRIException("IO %s is not an output value.", name);
212  }
213 
214  // add IO value to command message
215  FriIOValue& ioValue = cmdData.writeIORequest[cmdData.writeIORequest_count];
216 
217  strncpy(ioValue.name, name, sizeof(ioValue.name) - 1);
218  ioValue.name[sizeof(ioValue.name) - 1] = 0; // ensure termination
219  ioValue.type = ioType;
220  ioValue.has_digitalValue = (ioType == FriIOType_DIGITAL | ioType == FriIOType_BOOLEAN);
221  ioValue.has_analogValue = (ioType == FriIOType_ANALOG);
222  ioValue.direction = FriIODirection_OUTPUT;
223 
224  cmdData.writeIORequest_count ++;
225  message->has_commandData = true;
226 
227  return ioValue;
228  }
229  else
230  {
231  throw FRIException("Exceeded maximum number of IOs that can be set.");
232  }
233  }
